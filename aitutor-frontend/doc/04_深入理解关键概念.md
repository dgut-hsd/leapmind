### 1. Three.js 基础概念

#### **1.1 三大核心对象**

```javascript
// Three.js的所有3D渲染都基于三个核心对象：

// 1️⃣ Scene（场景）- 容器
const scene = new THREE.Scene();
// 作用：存放所有3D对象（模型、灯光、相机等）
// 类比：舞台

// 2️⃣ Camera（相机）- 视角
const camera = new THREE.PerspectiveCamera(75, width / height, 0.1, 1000);
camera.position.set(0, 1.0, 0.7);  // 相机位置
// 作用：决定用户看到什么
// 类比：摄像机

// 3️⃣ Renderer（渲染器）- 输出
const renderer = new THREE.WebGLRenderer({ canvas });
renderer.setSize(width, height);
// 作用：将3D场景渲染到canvas上
// 类比：电视屏幕
```

#### **1.2 渲染循环**

```javascript
// Three.js需要一个连续的渲染循环来显示动画

function animate() {
  requestAnimationFrame(animate);  // 60fps
  
  // 更新对象位置/旋转/缩放
  model.position.x += 0.01;
  model.rotation.y += 0.05;
  
  // 渲染到屏幕
  renderer.render(scene, camera);
}

animate();
```

**为什么需要循环？**
- 每帧（约16.67ms）需要更新对象状态
- 每帧需要重新渲染一次
- 人眼看起来就是连续的动画

#### **1.3 灯光系统**

### 2. VRM 格式详解

#### **2.1 什么是VRM？**

```
VRM = Virtual Reality Model
= 虚拟现实模型的标准格式

特点：
✅ 开放标准
✅ 基于GLTF/GLB格式
✅ 包含人形骨骼信息
✅ 包含表情BlendShape
✅ 支持第一人称视角
✅ 支持注视系统
✅ 跨平台兼容
```

#### **2.2 VRM的三个核心组件**

```javascript
// 加载VRM后得到的对象结构：

const vrm = gltf.userData.vrm;

// 1️⃣ humanoid - 人形骨骼系统
vrm.humanoid.getNormalizedBoneNode(VRMHumanBoneName.Head);
vrm.humanoid.getNormalizedBoneNode(VRMHumanBoneName.LeftArm);
vrm.humanoid.getNormalizedBoneNode(VRMHumanBoneName.Spine);

// 用途：用于骨骼动画（改变骨骼的旋转/位置）
// 优势：骨骼很少，计算快；还能自动计算其他部分

// 2️⃣ expressions - 表情系统（BlendShape）
vrm.expressionManager.setValue("happy", 1.0);  // happy表情100%
vrm.expressionManager.setValue("sad", 0.0);     // sad表情0%
vrm.expressionManager.setValue("angry", 0.5);   // angry表情50%

// 用途：混合不同表情，可以有中间状态
// 优势：表情自然，支持混合

// 3️⃣ lookAt - 注视系统
vrm.lookAtTargetParent.position.set(0, 0, 1);
// 虚拟人物的眼睛会跟随这个目标点

// 用途：让虚拟人物看向特定方向
// 优势：自动计算眼睛转动，看起来很自然
```

#### **2.3 BlendShape系统详解**

```
BlendShape = 混合形状
原理：在两个或多个模型形状之间进行插值

例如：中性表情 → 笑脸
方法：
1. 定义两个不同的网格形状
   - neutral: 原始嘴形
   - smile: 嘴角上扬的嘴形

2. 使用权重0-1来混合
   - weight = 0.0 → 纯neutral
   - weight = 0.5 → 50% neutral + 50% smile（半笑）
   - weight = 1.0 → 纯smile

3. 应用：
   vrm.expressionManager.setValue("happy", 0.7);
   
这样就能实现自然的表情变化！
```

### 3. 音频处理基础

#### **3.1 Web Audio API 简介**

```javascript
// Web Audio API 是浏览器的音频处理库

// 核心概念：
const audioContext = new (window.AudioContext || window.webkitAudioContext)();

// 音频处理链：
输入(音频文件/麦克风) 
  → AudioSource(音频源)
  → 分析器(AnalyserNode)
  → 音量控制(GainNode)
  → 输出(destination/扬声器)

// 代码示例：
const source = audioContext.createBufferSource();
const analyser = audioContext.createAnalyser();
const gain = audioContext.createGain();

source.connect(analyser);      // 连接分析器
analyser.connect(gain);         // 连接音量控制
gain.connect(audioContext.destination);  // 连接输出

source.start();  // 开始播放
```

#### **3.2 FFT 频谱分析**

```javascript
// FFT = Fast Fourier Transform（快速傅里叶变换）
// 将时域信号转换为频域信号

// 时域 - 声波波形
时间 (秒) →
│    /\      /\
│   /  \    /  \
└──────────────────→ 音量

// 频域 - 各频率成分
频率 (Hz) →
│     △
│    △ △
│   △   △
└──────────────────→ 幅度

// 代码：
const frequencyData = new Uint8Array(analyser.frequencyBinCount);
analyser.getByteFrequencyData(frequencyData);

// frequencyData[0] = 低频分量
// frequencyData[中间] = 中频分量
// frequencyData[末尾] = 高频分量

// 应用：
// - 低频强 → 可能是元音"A"或"O"
// - 高频强 → 可能是辅音或"S"
// - 这就是LipSync的原理！
```

#### **3.3 时域分析**

```javascript
// 时域 = 声波在时间上的变化

const timeDomainData = new Float32Array(analyser.fftSize);
analyser.getFloatTimeDomainData(timeDomainData);

// 计算RMS（均方根）- 整体音量
let rms = 0;
for (let i = 0; i < timeDomainData.length; i++) {
  rms += timeDomainData[i] * timeDomainData[i];
}
rms = Math.sqrt(rms / timeDomainData.length);

// 计算Peak（峰值）- 最大瞬间音量
let peak = Math.max(...timeDomainData.map(Math.abs));

// 这两个值用于：
// 1. 计算嘴巴张开程度
// 2. 检测是否有人在说话
// 3. 进行包络检测
```

### 4. 旋转和变换

#### **4.1 Euler角 vs 四元数**

```javascript
// Euler角（欧拉角）- 直观但有万向锁问题
const euler = new THREE.Euler(
  Math.PI / 4,  // x轴旋转（俯仰）
  Math.PI / 3,  // y轴旋转（偏航）
  Math.PI / 6   // z轴旋转（滚动）
);

// 在headMotionController中使用Euler角：
bigNod: {
  keyframes: [
    { rotation: new THREE.Euler(0, 0, 0) },
    { rotation: new THREE.Euler(-0.4, 0, 0) },  // 低头
    { rotation: new THREE.Euler(0.1, 0, 0) },   // 中途
  ]
}

// 三个分量的含义：
// x > 0: 抬头, x < 0: 低头
// y > 0: 向右转, y < 0: 向左转
// z > 0: 右倾, z < 0: 左倾
```

#### **4.2 插值动画原理**

```javascript
// 关键帧动画的核心：在两个关键帧之间插值

// 例如：大幅点头
keyframes = [
  { time: 0.0, rotation: Euler(0, 0, 0) },      // 开始
  { time: 0.3, rotation: Euler(-0.4, 0, 0) },   // 最低点
  { time: 1.2, rotation: Euler(0, 0, 0) }       // 结束
]

// 当前时间 = 0.15秒（中途）
// 进度 = 0.15 / 1.2 = 0.125（12.5%）

// 找到相邻的两个关键帧：
frame1 = { time: 0.0, rotation: Euler(0, 0, 0) }
frame2 = { time: 0.3, rotation: Euler(-0.4, 0, 0) }

// 在这两帧之间的本地进度：
localProgress = (0.15 - 0.0) / (0.3 - 0.0) = 0.5

// 应用平滑函数（smoothstep）：
smoothProgress = localProgress * localProgress * (3 - 2 * localProgress)
                = 0.5 * 0.5 * (3 - 2 * 0.5)
                = 0.25 * 2
                = 0.5  (这样会更平滑)

// 插值旋转：
current.x = lerp(0, -0.4, smoothProgress) = 0 + (-0.4) * 0.5 = -0.2
current.y = lerp(0, 0, smoothProgress) = 0
current.z = lerp(0, 0, smoothProgress) = 0

// 应用到骨骼：
headBone.rotation = Euler(-0.2, 0, 0)
```

#### **4.3 Smoothstep函数**

```javascript
// Smoothstep = 平滑阶跃函数
// 使动画加速度变化平滑，看起来更自然

function smoothstep(t) {
  // t 是 0-1 之间的值
  return t * t * (3 - 2 * t);
}

// 比较：
linear:    y = t        // 直线运动，生硬
smoothstep: y = t²(3-2t) // 平滑运动，自然

// 动画曲线：
linear:    ╱
           /
smoothstep: ╭
            ╯  // 两端缓，中间快

// 这就是为什么虚拟人物的动作看起来很自然！
```

### 6. 关键词匹配算法

#### **6.1 HeadMotionController中的关键词检测**

```javascript
// 源代码位置：headMotionController.js 第328-348行

_detectMotionFromText(text) {
  const content = text.trim();
  if (!content) return null;

  // 优先级顺序：从强到弱
  const order = [
    "bigShake",     // 最优先：强烈否定
    "bigNod",       // 其次：强烈肯定
    "lookUp",       // 惊讶
    "tiltHead",     // 困惑
    "smallShake",   // 轻微否定
    "smallNod",     // 轻微肯定
  ];

  // 按优先级顺序检查
  for (const type of order) {
    const list = this._triggers[type];  // 获取该动作的关键词列表
    if (list && list.some((kw) => content.includes(kw))) {
      return type;  // 找到匹配立即返回
    }
  }
  return null;  // 未找到任何匹配
}

// 为什么这样设计？
// ✅ 优先级：strong signal优先
// ✅ 效率：find-first机制，快速返回
// ✅ 可配置：DEFAULT_TRIGGERS可以修改
```

### 7. 表情-动作映射

#### **7.1 两个映射关系**

```javascript
// 映射1：动作 → 表情
const MOTION_TO_EXPRESSION = {
  bigNod: "happy",       // 大幅点头自动显示开心
  smallNod: "relaxed",   // 轻微点头显示放松
  tiltHead: "surprised", // 歪头显示惊讶
  bigShake: "angry",     // 大幅摇头显示愤怒
  smallShake: "sad",     // 小幅摇头显示悲伤
  lookUp: "surprised",   // 抬头显示惊讶
};

// 映射2：情绪 → 可能的动作
const EMOTION_TO_MOTION = {
  happy: ["smallNod", "lookUp"],        // 开心倾向点头或抬头
  angry: ["bigShake", "smallShake"],    // 愤怒倾向摇头
  sad: ["smallShake", "tiltHead"],      // 悲伤倾向摇头或歪头
  surprised: ["lookUp", "tiltHead"],    // 惊讶倾向抬头或歪头
  // ...
};

// 为什么双向映射？
// ✅ 从动作触发表情（自动配套）
// ✅ 从表情选择动作（反向）
// ✅ 增强虚拟人物的表现力
```

#### **7.2 动作播放流程**

```javascript
playMotionByContent(message, emotionOrStyle) {
  // 优先级1：使用情绪/风格
  if (emotionOrStyle && EMOTION_TO_MOTION[emotionOrStyle]?.length) {
    this.playMotionByEmotion(emotionOrStyle);
    return;
  }
  
  // 优先级2：使用关键词
  const type = this._detectMotionFromText(message);
  if (type) {
    this.playHeadMotion(type);
  }
}

playHeadMotion(motionType) {
  // Step1: 从MOTION_TO_EXPRESSION查找对应表情
  const expression = MOTION_TO_EXPRESSION[motionType];
  if (this._expressionController && expression) {
    // Step2: 播放表情
    this._expressionController.playEmotion(expression);
  }
  
  // Step3: 记录动作（供update()使用）
  this._currentMotion = {
    type: motionType,
    startTime: this._clock.getElapsedTime(),
    duration: this._motionDatabase[motionType].duration,
    keyframes: this._motionDatabase[motionType].keyframes
  };
}

// 时间线：
0ms:     playHeadMotion("bigNod") 被调用
         ↓
1ms:     playEmotion("happy") 立即执行（改变BlendShape权重）
         ↓
2ms:     记录_currentMotion（下面update()会用）
         ↓
16ms:    update()被调用（下一帧）
         ├─ 计算动作进度（16-0) / 1200 = 0.013
         ├─ 找到关键帧区间
         ├─ 插值计算当前旋转
         └─ 应用到headBone
         ↓
32ms:    renderer.render() 渲染（看到虚拟人物点头并显示开心表情）
```
